## Install

```bash
npm install --save react-charts-builder
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-charts-builder'
import 'react-charts-builder/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
